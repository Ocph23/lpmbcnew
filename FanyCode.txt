Berikut adalah contoh implementasi aplikasi penjamin mutu universitas menggunakan Laravel sebagai backend dan Vue.js sebagai frontend:

Backend (Laravel)

Instalasi Dependensi
Buat projek baru dengan laravel new dan instal dependensi yang dibutuhkan:

composer require --prefer-dist barryvdh/laravel-debugbar "^3.5"
composer require --prefer-dist laravelcollective/html "^6.2"
Model dan Eloquent
Buat model untuk Dokumen, Bukti Kegiatan, dan Tim Auditor:

// app/Models/Dokumen.php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Relations\HasMany;

class Dokumen extends Model
{
    protected $fillable = ['judul', 'konten'];

    public function bukti_kegiatan(): HasMany
    {
        return $this->hasMany(BuktiKegiatan::class);
    }
}

// app/Models/BuktiKegiatan.php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Relations\BelongsTo;

class BuktiKegiatan extends Model
{
    protected $fillable = ['judul', 'konten', 'dokumen_id'];

    public function dokumen(): BelongsTo
    {
        return $this->belongsTo(Dokumen::class);
    }
}

// app/Models/TimAuditor.php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Relations\HasMany;

class TimAuditor extends Model
{
    protected $fillable = ['nama', 'jabatan'];

    public function audit(): HasMany
    {
        return $this->hasMany(Audit::class);
    }
}
Kontroler dan Route
Buat kontroler untuk Dokumen, Bukti Kegiatan, Tim Auditor, dan Audit:

// app/Http/Controllers/DokumenController.php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use App\Models\Dokumen;

class DokumenController extends Controller
{
    public function index()
    {
        $dokumens = Dokumen::all();
        return view('dokumen.index', compact('dokumens'));
    }

    public function create()
    {
        return view('dokumen.create');
    }

    public function store(Request $request)
    {
        $dokumen = new Dokumen($request->all());
        $dokumen->save();
        return redirect()->route('dokumen.index');
    }

    public function show(Dokumen $dokumen)
    {
        return view('dokumen.show', compact('dokumen'));
    }

    public function edit(Dokumen $dokumen)
    {
        return view('dokumen.edit', compact('dokumen'));
    }

    public function update(Request $request, Dokumen $dokumen)
    {
        $dokumen->update($request->all());
        return redirect()->route('dokumen.index');
    }

    public function destroy(Dokumen $dokumen)
    {
        $dokumen->delete();
        return redirect()->route('dokumen.index');
    }
}

Migrasi
Buat migrasi untuk Dokumen, Bukti Kegiatan, Tim Auditor, dan Audit:

php artisan make:migration create_dokumens_table

php artisan make:migration create_buktikegitan_table

php artisan make:migration create_tim_auditors_table

php artisan make:migration create_audit_table
Lakukan migrasi:

php artisan migrate
Seeder
Buat seeder untuk Dokumen, Bukti Kegiatan, Tim Auditor, dan Audit:

php artisan make:seeder DokumenSeeder

php artisan make:seeder BuktiKegiatanSeeder

php artisan make:seeder TimAuditorSeeder

php artisan make:seeder AuditSeeder
Lakukan seeder:

php artisan db:seed --class=DokumenSeeder

php artisan db:seed --class=BuktiKegiatanSeeder

php artisan db:seed --class=TimAuditorSeeder

php artisan db:seed --class=AuditSeeder
Vue.js
Buat komponen untuk Dokumen, Bukti Kegiatan, Tim Auditor, dan Audit:

// src/components/Dokumen.vue

<template>
  <div>
    <h1>Dokumen</h1>
    <ul>
      <li v-for="dokumen in dokumens" :key="dokumen.id">
        {{ dokumen.judul }}
      </li>
    </ul>
  </div>
</template>

<script>
export default {
  data() {
    return {
      dokumens: []
    };
  },
  mounted() {
    this.$axios.get('/api/dokumen')
      .then(response => {
        this.dokumens = response.data;
      })
      .catch(error => {
        console.error(error);
      });
  }
};
</script>

Main.js
Buat main.js untuk mengimport komponen:

// src/main.js

import Vue from 'vue';
import App from './App.vue';

Vue.component('Dokumen', () => import('./components/Dokumen.vue'));
Vue.component('BuktiKegiatan', () => import('./components/BuktiKegiatan.vue'));
Vue.component('TimAuditor', () => import('./components/TimAuditor.vue'));
Vue.component('Audit', () => import('./components/Audit.vue'));

new Vue({
  render: h => h(App)
}).$mount('#app');

Buat app.vue untuk menampilkan komponen:

Pengujian
Buat pengujian untuk memastikan bahwa aplikasi berjalan dengan baik:

php artisan test
Pengujian akan memeriksa apakah semua komponen, kontroler api, dan route api berfungsi dengan benar.

Dengan demikian, aplikasi penjamin mutu universitas menggunakan Laravel sebagai backend dan Vue.js sebagai frontend sudah siap digunakan.
